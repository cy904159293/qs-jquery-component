基本使用
基本使用，尝试直接在input中输入日期。 
new DatePicker({
    input:$('#test1'),
    onDateChange:(date)=>{
        message({
            type:'success',
            msg:'当前选中日期：'+date
        })
    }
});
定义不同格式的日期
使用defaultFormat参数，定义不同格式的日期。 
new DatePicker({
    input:$('#test2'),
    defaultFormat:'yyyy/MM/dd',
    currentDate:Date.now(),
    onDateChange:(date)=>{
        message({
            type:'success',
            msg:'当前选中日期：'+date
        })
    }
});

new DatePicker({
    input:$('#test3'),
    defaultFormat:'yyyy年MM月dd日',
    currentDate:Date.now(),
    onDateChange:(date)=>{
        message({
            type:'success',
            msg:'当前选中日期：'+date
        })
    }
});
不同类型的日期选择
使用type参数，定义不同类型的日期选择。 
new DatePicker({
    input:$('#test4'),
    type:'year',
    defaultFormat:'yyyy年',
    onDateChange:(date)=>{
        message({
            type:'success',
            msg:'当前选中日期：'+date
        })
    }
});

new DatePicker({
    input:$('#test5'),
    type:'month',
    defaultFormat:'yyyy年MM月',
    onDateChange:(date)=>{
        message({
            type:'success',
            msg:'当前选中日期：'+date
        })
    }
});

区间选择
使用两个组件组合来进行区间选择，通过setStartDate和通过setEndDate和确保选择区间。 
this.startDatePicker = new DatePicker({
    input:$('#test6'),
    onDateChange:(date)=>{
        this.endDatePicker.setStartDate(date).pickerShow();
    }
});

this.endDatePicker = new DatePicker({
    input:$('#test7'),
    onDateChange:(date)=>{
        this.startDatePicker.setEndDate(date)
    }
});
