基本使用
基本使用。 
new Rate({
    outer:$('#test1'),
    readOnly:false
});
只读评分，允许半星
只读评分，设置readOnly为true；通过currentRate来设置当前星数，允许半星。 
new Rate({
    outer:$('#test2'),
    currentRate:2
});

new Rate({
    outer:$('#test3'),
    currentRate:3.4
});
显示不同颜色和文字
通过colorCB和textCB两个回调函数来，显示不同的颜色和文字。 
new Rate({
    outer:$('#test4'),
    readOnly:false,
    colorCB:n=>{
        n = parseInt(n);
        const color = [ '#ea6345', '#ea6345', '#aa6e92', '#4585f3', '#16a4c8', '#00bf8b' ];
        return color[n];
    },
    textCB:n=>{
        if(n === 0) return '';
        if(n<2) return '差';
        if(n<4) return '一般';
        return '很好'
    }
});
设置只读评分
通过setStar来设置只读评分的星数。 
this.rate = new Rate({
outer:$('#test5'),
colorCB:n=>{
    n = parseInt(n);
    const color = [ '#ea6345', '#ea6345', '#aa6e92', '#4585f3', '#16a4c8', '#00bf8b' ];
    return color[n];
},
textCB:n=>{
    if(n === 0) return '';
    if(n<2) return '差';
    if(n<4) return '一般';
    return '很好'
}
});

const that = this;
const rate = $(this).prev().find('input').val();
that.rate.setStar(rate);
message({
    type:'success',
    msg:'当前评分为'+rate
})
